\section*{Examples of Cost Functions Satisfying QI}

\begin{itemize}
\item $w(j,i)=b[j]\cdot a[i]$, provided that for all $i<j$, $a[i]<a[j]$ and $b[i]>b[j]$.
\item $w(j,i)=(x[i]-x[j])^2 + C$, provided that for all $j<i$, $x[j]<x[i]$.
\item $w(j,i)=b[j]a[i] + b[j]^2 a[i]^2 + b[j]^3 a[i]^3$, provided that for all $i<j$, $0<a[i]<a[j]$ and $b[i]>b[j]>0$.
\end{itemize}
